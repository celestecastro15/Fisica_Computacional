1. Realice una función que tenga como entrada los puntos {$x_i$,$f_i$} y que regrese {$x_i$, $F_i$}, donde $F_i$ representa la integral de la función hasta el $x_{i-ésimo}$ punto. Para corroborar que su función trabaja de manera adecuada, puede utilizar la función cos(x) en el intervalo [0, 2π]. Puede utilizar cualquier algoritmo de integración que vimos en clase o en las tareas.¶

Vamos a utilizar el algoritmo de integración desarrollado en el problema 1a) de la Tarea 2, el cual tiene la siguiente expresión:

$$ \int_{x_i -h}^{x_i+h} f(x) dx = \frac{f(x_{i-1})+4 f(x_i) + f(x_{i+1})}{3} h $$

Y para el primer intervalo:

$$ \frac{h}{48} (23 f_1 + 26f_2 - 4f_3 + 6f_4 -3f_5) $$

Ahora, observamos que el algoritmo anterior funciona a partir de 5 puntos ya que para calcular el 1er intervalo requiere los primeros 5 puntos, por tanto, tenemos que considerar por separado los casos para 1,2,3 y 4 puntos.

A continuación procedemos a programar el algoritmo requerido:

In [6]:
function int_array(x,f)
    I=zeros(length(f)) #I será un arreglo que vamos a ir llenando con el valor de la integral hasta el punto xi
    I[1]=0 #La integral hasta el punto 1 es cero porque no hay un intervalo
    
    #Vamos a calcular la integral para para 2,3 y 4 puntos por medio de rectángulos siguiendo la idea de la suma de Riemann
    
    h=x[2]-x[1] #Como se tiene una malla uniforme la longitud de los intervalos siempre es la misma 
    
    I[2]=f[2]*h
    I[3]=f[3]*h + I[2]
    I[4]=f[4]*h + I[3]
    
    #A partir de aquí ya utilizamos el algoritmo de la Tarea 2
    #Recordamos que en dicha tarea vimos que tenemos que dividir en dos casos: Par o impar
    #La malla va a ir terminando en el i-ésimo elemento, porque justo buscamos la integral hasta el punto xi
    for i in range(5, stop=length(f))
        
        if mod((i),2) == 1 #Caso impar
            for j in range(2,stop=i-1,step=2) 
                I[i] += (f[j+1] + 4*f[j] + f[j-1])*(h/3)
            end
        end
        
        if mod((i),2) == 0 #Caso par
            integral=0 #Definimos la integral 
            for j in range(3,stop=i-1,step=2)
                integral += (f[j+1] + 4*f[j] + f[j-1])*(h/3)
                #Ahora, tenemos que irle sumando a cada integral la expresión para el 1er intervalo que es el que falta cuando estamos en el caso par
                I[i]=integral+(h/48)*(23*f[1] + 26*f[2] - 4*f[3] + 6*f[4] -3*f[4]) 
            end
        end
    end
    return I
end

function deriv_h3(f,x)
    f_prima = zeros(length(f))
    h = x[2] - x[1]
    dosh = 2*h
    for i in 2:(length(f)-1)
        f_prima[i] = ( f[i+1]-f[i-1] )/(dosh)
    end
    # calcular el primer punto
    f_prima[1] = ( -0.5*f[3] + 2*f[2] -1.5*f[1] )/h
    # calcular el ultimo punto
    f_prima[end] = -( -0.5*f[end-2] + 2*f[end-1] -1.5*f[end] )/h    
    return f_prima
end

function deriv_h5(f,x)
    f_prima = zeros(length(f))
    h = x[2] - x[1]
    doceh = 12*h
    for i in 3:(length(f)-2)
        f_prima[i] = ( -f[i+2]+ 8*(f[i+1]-f[i-1]) +f[i-2] )/(doceh)
    end
    c0,c1,c2,c3,c4 = -25/12,4,-3,4/3,-0.25
    # calcular el primer punto
    f_prima[1] = ( c0*f[1] + c1*f[2] + c2*f[3] + c3*f[4] + c4*f[5] )/h
    # calcular el segundo punto
    f_prima[2] = ( c0*f[2] + c1*f[3] + c2*f[4] + c3*f[5] + c4*f[6] )/h
    # calcular el ultimo punto o el punto N
    f_prima[end] = -( c0*f[end] + c1*f[end-1] + c2*f[end-2] + c3*f[end-3] + c4*f[end-4] )/h
    # calcular el penultimo punto o el N-1
    f_prima[end-1] = -( c0*f[end-1] + c1*f[end-2] + c2*f[end-3] + c3*f[end-4] + c4*f[end-5] )/h
    return f_prima
end

function int_alg2(x,y)

        inicio = 3
    elseif mod(length(y),2) == 1

end

function int_num_boole(x,y) #x-malla de puntos, y-función a la cual le vamos a calcular su integral
    integral = 0 #Definimos la integral para poder utilizarla más adelante
    h = x[2] - x[1] #h siempre tendrá el mismo valor porque se trata de una malla uniforme
    
    if mod(length(y),4) == 1 #No sobra ningún punto
        inicio = 1
    end
    if mod(length(y),4) == 2 #Sobra 1 punto
        inicio = 2
    end
    
    if mod(length(y),4) == 3 #Sobran 2 puntos
        inicio = 3
    end
    
    if mod(length(y),4) == 0 #Sobran 3 puntos
        inicio = 4
    end
    
    for i in inicio:4:length(y)-4 #Avanza de 4 en 4 y termina en N-4
        integral += 7*y[i] + 32*y[i+1] + 12*y[i+2] + 32*y[i+3] + 7*y[i+4]
    end
    integral = integral*(2*h/45)

    #para agregar lo correspondiente al 1er intervalo, a los primeros 2 intervalos 
    #ó a los primeros 3 intervalos. 
    
    #Expresión para el 1er intervalo

        integral += (h/48)*(23*y[1] + 26*y[2] - 4*y[3] + 6*y[4] -3*y[5])

    return integral
end

3. Sabemos que $\int_1^{\infty} \frac{dx}{x^2} = 1$, sin embargo numéricamente no es factible utilizar el infinito. De esta manera genere una forma de calcular dicha integral hasta obtener una precisión de $10^{−3}$. Esto se puede generalizar siguiendo la misma idea para los casos en donde el límite inferior de la integral también es infinito, esto podría hacerse partiendo de un valor inicial para ese límite, y mediante otro ciclo for ir variándolo de igual manera que va variando el límite superior. Asimismo, funciona para todas las funciones pues solo es necesario definirlas dentro del algoritmo que calcula la integral. Finalmente, para el caso donde los límites no son necesariamente infinito pero la función se indetermina, podemos hacer lo mismo, partir de un valor incial e ir variándolo para irnos acercando al valor del límite.

Para ejemplificar un poco esta generalización, podemos definir una función que calcule la integral que va desde un cierto número hasta infinito, para cualquier función:

function int_lim(resultado,inicio,num_puntos,error,y) 
    #resultado-resultado esperado de la integral
    #inicio-número donde inicia el intervalo de integración
    #num_puntos-número de puntos que queremos que tenga nuestra malla
    #error- orden del máximo error que vamos a aceptar
    #y-función
    
    i=2
    x = collect(range(inicio,stop=i,length=num_puntos)) 

    #Vamos a generar un ciclo while que vaya aumentando la i para ir aumentando el intervalo de integración
    #Este ciclo se va a repetir mientras el error siga siendo mayor al error máximo que queremos

    while abs(resultado-int_alg2(x,y))>=error
        i +=1 #i va a ir aumentando en cada ciclo
        x=collect(range(inicio,stop=i,length=num_puntos)) 
    end

#Ahora, como ya termino el ciclo while, i ya tiene el valor adecuado que arroja un valor para la integral con un error menor que el error máximo
    x=collect(range(inicio,stop=i,length=num_puntos))
    println("El valor de la integral es: ",int_alg2(x,x.^(-2)))
    println("El error es: ", abs(1-int_alg2(x,x.^(-2))))
end
